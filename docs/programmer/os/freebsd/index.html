    换源 #

    mkdir -p /usr/local/etc/pkg/repos
vim /usr/local/etc/pkg/repos/bjtu.conf

    bjtu: {
    url: "pkg+http://mirror.bjtu.edu.cn/reverse/freebsd-pkg/${ABI}/quarterly",
    mirror_type: "srv",
    signature_type: "none",
    fingerprints: "/usr/share/keys/pkg",
    enabled: yes
}
FreeBSD: { enabled: no }

    pkg update
